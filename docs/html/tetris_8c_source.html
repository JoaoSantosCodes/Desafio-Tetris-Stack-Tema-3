249int calcularPontuacao(char tipoPeca, SistemaExpert* sistemaPtr) {
250 int pontosBase = 0;
251
252 // ═══════════════════════════════════════════════════════════════
253 // PONTUAÇÃO BASE POR TIPO DE PEÇA
254 // ═══════════════════════════════════════════════════════════════
255 // Cada tipo de peça tem valor diferente baseado na complexidade
256 switch (tipoPeca) {
257 case 'I': pontosBase = 100; break; // Linha - mais comum, menor valor
258 case 'O': pontosBase = 120; break; // Quadrado - versátil, valor médio
259 case 'T': pontosBase = 150; break; // T - estratégico, valor alto
260 case 'L': pontosBase = 180; break; // L - mais complexo, maior valor
261 default: pontosBase = 50; break; // Fallback para tipos inválidos
262 }
263
264 // ═══════════════════════════════════════════════════════════════
265 // APLICAÇÃO DE MULTIPLICADORES
266 // ═══════════════════════════════════════════════════════════════
267 // Fórmula: PontosBase × MultiplicadorCombo × FatorDificuldade
268 int pontosFinais = (int)(pontosBase * sistemaPtr->multiplicadorAtual * sistemaPtr->fatorDificuldade);
269
270 return pontosFinais;
271}

285int detectarCombo(char tipoPeca, SistemaExpert* sistemaPtr) {
286 int comboFormado = 0;
287
288 // ═══════════════════════════════════════════════════════════════
289 // VERIFICAÇÃO DE SEQUÊNCIA
290 // ═══════════════════════════════════════════════════════════════
291 if (sistemaPtr->ultimoTipoJogado == tipoPeca) {
292 // Continua sequência do mesmo tipo de peça
293 sistemaPtr->sequenciaTipoAtual++;
294
295 // ═══════════════════════════════════════════════════════════════
296 // FORMAÇÃO DE COMBO
297 // ═══════════════════════════════════════════════════════════════
298 // Combo é formado a partir de 3 peças iguais consecutivas
299 if (sistemaPtr->sequenciaTipoAtual >= 3) {
300 sistemaPtr->comboAtual++; // Incrementa combo atual
301 sistemaPtr->totalCombos++; // Incrementa total de combos
302 comboFormado = 1; // Sinaliza que combo foi formado
303
304 // Atualiza recorde de melhor combo da sessão
305 if (sistemaPtr->comboAtual > sistemaPtr->melhorCombo) {
306 sistemaPtr->melhorCombo = sistemaPtr->comboAtual;
307 }
308
309 // ═══════════════════════════════════════════════════════════════
310 // AUMENTO DO MULTIPLICADOR
311 // ═══════════════════════════════════════════════════════════════
312 // Multiplicador aumenta com combos (máximo 10x)
313 if (sistemaPtr->multiplicadorAtual < 10) {
314 sistemaPtr->multiplicadorAtual++;
315 }
316 }
317 } else {
318 // ═══════════════════════════════════════════════════════════════
319 // QUEBRA DE SEQUÊNCIA
320 // ═══════════════════════════════════════════════════════════════
321 // Tipo diferente quebra a sequência - reset dos combos
322 sistemaPtr->comboAtual = 0; // Zera combo atual
323 sistemaPtr->sequenciaTipoAtual = 1; // Inicia nova sequência
324 sistemaPtr->multiplicadorAtual = 1; // Reset multiplicador
325 }
326
327 // Atualiza cache do último tipo jogado para próxima verificação
328 sistemaPtr->ultimoTipoJogado = tipoPeca;
329 return comboFormado;
330}

354int verificarProgressaoNivel(SistemaExpert* sistemaPtr) {
355 // ═══════════════════════════════════════════════════════════════
356 // VERIFICAÇÃO DE PROGRESSÃO
357 // ═══════════════════════════════════════════════════════════════
358 // Verifica se atingiu pontuação necessária para próximo nível
359 if (sistemaPtr->pontuacaoNivel >= sistemaPtr->limitePontosNivel) {
360
361 // ═══════════════════════════════════════════════════════════════
362 // AVANÇO DE NÍVEL
363 // ═══════════════════════════════════════════════════════════════
364 sistemaPtr->nivelAtual++; // Incrementa nível atual
365 sistemaPtr->pontuacaoNivel = 0; // Reset pontuação do nível
366
367 // ═══════════════════════════════════════════════════════════════
368 // AJUSTE DE DIFICULDADE PROGRESSIVA
369 // ═══════════════════════════════════════════════════════════════
370 // Aumenta dificuldade progressivamente (máximo 3.0x)
371 if (sistemaPtr->fatorDificuldade < 3.0) {
372 sistemaPtr->fatorDificuldade += 0.2; // Incremento de 20% por nível
373 }
374
375 // ═══════════════════════════════════════════════════════════════
376 // CÁLCULO EXPONENCIAL DO PRÓXIMO NÍVEL
377 // ═══════════════════════════════════════════════════════════════
378 // Fórmula: 1000 * nível² (progressão quadrática)
379 // Nível 1: 1.000 pts | Nível 2: 4.000 pts | Nível 3: 9.000 pts
380 sistemaPtr->limitePontosNivel = 1000 * sistemaPtr->nivelAtual * sistemaPtr->nivelAtual;
381 sistemaPtr->pontosParaProximoNivel = sistemaPtr->limitePontosNivel;
382
383 return 1; // Confirmação: subiu de nível
384 }
385
386 // ═══════════════════════════════════════════════════════════════
387 // ATUALIZAÇÃO DE PONTOS RESTANTES
388 // ═══════════════════════════════════════════════════════════════
389 // Calcula quantos pontos faltam para o próximo nível
390 sistemaPtr->pontosParaProximoNivel = sistemaPtr->limitePontosNivel - sistemaPtr->pontuacaoNivel;
391
392 return 0; // Não houve progressão de nível
393}

481 // ═══════════════════════════════════════════════════════════════

601 // VALIDAÇÃO DE INTEGRIDADE
602 // ═══════════════════════════════════════════════════════════════

606 if (sistemaPtr->pontuacaoNivel < 0) sistemaPtr->pontuacaoNivel = 0;

612 if (sistemaPtr->nivelAtual > 10) sistemaPtr->nivelAtual = 10;

614 // ═══════════════════════════════════════════════════════════════
615 // OTIMIZAÇÃO DE CÁLCULOS
616 // ═══════════════════════════════════════════════════════════════
617
618 // Recalcular eficiência da reserva
619 if (sistemaPtr->totalJogadas > 0) {
620 int usoReserva = (sistemaPtr->jogadasDaPilha * 100) / sistemaPtr->totalJogadas;

627

637 maxContagem = sistemaPtr->contagemTipoT;

642 tipoMax = 'L';

653 if (sistemaPtr->fatorDificuldade > 3.0) sistemaPtr->fatorDificuldade = 3.0;

661}
